searchtype	Utilisation
contains	Contient une valeur
equals	Égal à
notequals	Différent de
lessthan	Inférieur à
morethan	Supérieur à
under	Sous une entité/catégorie hiérarchique
notunder	Pas sous une entité/catégorie hiérarchique

criteria[0][field]=12
&criteria[0][searchtype]=equals
&criteria[0][value]=2
